Quick Start#

Queries#

 1import opcc
 2import numpy as np
 3from sklearn import metrics
 4
 5env_name = 'HalfCheetah-v2'
 6dataset_name = 'random'
 7
 8# ########################################################
 9# Policy Comparison Queries (PCQ) (Section : 3.1 in paper)
10# ########################################################
11# Queries are dictionaries with policies as keys and
12# corresponding queries as values.
13queries = opcc.get_queries(env_name)
14
15
16def random_predictor(obs_a, obs_b, action_a, action_b,
17                     policy_a, policy_b, horizon):
18    answer = np.random.randint(low=0, high=2, size=len(obs_a)).tolist()
19    confidence = np.random.rand(len(obs_a)).tolist()
20    return answer, confidence
21
22
23targets = []
24predictions = []
25confidences = []
26# Batch iteration through Queries :
27for (policy_a_id, policy_b_id), query_batch in queries.items():
28    # retrieve policies
29    policy_a, _ = opcc.get_policy(*policy_a_id)
30    policy_b, _ = opcc.get_policy(*policy_b_id)
31
32    # query-a
33    obs_a = query_batch['obs_a']
34    action_a = query_batch['action_a']
35
36    # query-b
37    obs_b = query_batch['obs_b']
38    action_b = query_batch['action_b']
39
40    # horizon for policy evaluation
41    horizon = query_batch['horizon']
42
43    # ground truth binary vector:
44    # (Q(obs_a, action_a, policy_a, horizon)
45    # <  Q(obs_b, action_b, policy_b, horizon))
46    target = query_batch['target'].tolist()
47    targets += target
48
49    # Let's make predictions for the given queries.
50    # One can use any mechanism to predict the corresponding
51    # answer to queries, and we simply use a random predictor
52    # over here for demonstration purposes
53    p, c = random_predictor(obs_a, obs_b, action_a, action_b,
54                            policy_a, policy_b, horizon)
55    predictions += p
56    confidences += c

Evaluation Metrics#

 1# #########################################
 2# (Section 3.3 in paper)
 3# #########################################
 4loss = np.logical_xor(predictions, targets)  # we use 0-1 loss for demo
 5
 6# List of tuples (coverage, selective_risks, tau)
 7coverage_sr_tau = []
 8tau_interval=0.01
 9for tau in np.arange(0, 1 + 2 * tau_interval, tau_interval):
10  non_abstain_filter = confidences >= tau
11  if any(non_abstain_filter):
12    selective_risk = np.sum(loss[non_abstain_filter])
13    selective_risk /= np.sum(non_abstain_filter)
14    coverage = np.mean(non_abstain_filter)
15    coverage_sr_tau.append((coverage, selective_risk, tau))
16  else:
17    # 0 risk for 0 coverage
18    coverage_sr_tau.append((0, 0, tau))
19
20coverages, selective_risks, taus = list(zip(*sorted(coverage_sr_tau)))
21assert selective_risks[0] == 0 and coverages[0] == 0 , "no coverage not found"
22assert coverages[-1] == 1, 'complete coverage not found'
23
24# AURCC ( Area Under Risk-Coverage Curve): Ideally, we would like it to be 0
25aurcc = metrics.auc(x=coverages,y=selective_risks)
26
27# Reverse-pair-proportion
28rpp = np.logical_and(np.expand_dims(loss, 1)
29                     < np.expand_dims(loss, 1).transpose(),
30                     np.expand_dims(confidences, 1)
31                     < np.expand_dims(confidences, 1).transpose()).mean()
32
33# Coverage Resolution (cr_k) : Ideally, we would like it to be 1
34k = 10
35bins = [_ for _ in np.arange(0, 1, 1 / k)]
36cr_k = np.unique(np.digitize(coverages, bins)).size / len(bins)
37
38print("aurcc: {}, rpp: {}, cr_{}:{}".format(aurcc, rpp, k, cr_k))

Policy Usage#

 1import opcc, gym, torch
 2
 3env_name = "HalfCheetah-v2"
 4model, model_info = opcc.get_policy(env_name, pre_trained=1)
 5
 6done = False
 7env = gym.make(env_name)
 8
 9obs = env.reset()
10while not done:
11    action = model.actor(torch.tensor(obs).unsqueeze(0))
12    action = action.data.cpu().numpy()[0].astype('float32')
13    obs, reward, done, step_info = env.step(action)
14    env.render()
